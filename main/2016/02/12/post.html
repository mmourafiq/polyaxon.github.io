 7 It generates 36**6 = 2176782336 values (6 lowercase letters) which falls between 2**31 = 2147483648 and 2**32 = 4294967296.
 8 It pads the codes that are shorter than 6 characters with leading 'a' characters,

11         self.alphabet = alphabet

13         self.mask = (1 << length_generation) - 1
14         self.mapping = range(length_generation)
15         self.mapping.reverse()

18         return self.__enbase(self.__encode(n), min_length)

23     def __encode(self, n):
24         result = 0
25         for i, b in enumerate(self.mapping):
26             if (n & self.mask) & (1 << i):
27                 result |= (1 << b)
28         return (n & ~self.mask) | result
29 
30     def __decode(self, n):
31         result = 0
32         for i, b in enumerate(self.mapping):
33             if (n & self.mask) & (1 << b):
34                 result |= (1 << i)
35         return (n & ~self.mask) | result
36 
37     def __enbase(self, x, min_length=MIN_LENGTH):
38         result = self.__enbase_iter(x)
39         padding = self.alphabet[0] * (min_length - len(result))
40         return '%s%s' % (padding, result)
41 
42     def __enbase_iter(self, x):
43         n = len(self.alphabet)
44         if x < n:
45             return self.alphabet[x]
46         return self.__enbase_iter(x / n) + self.alphabet[x % n]
47 
48     def __debase(self, x):
49         n = len(self.alphabet)
50         result = 0
51         for i, c in enumerate(reversed(x)):
52             result += self.alphabet.index(c) * (n ** i)
53         return result
